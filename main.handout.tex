\documentclass[12pt, aspectratio=169, handout]{beamer}
\usepackage{pgfpages}
\pgfpagesdeclarelayout{4 on 1 new}
{
	\edef\pgfpageoptionheight{\the\paperheight} 
	\edef\pgfpageoptionwidth{\the\paperwidth}
	\edef\pgfpageoptionborder{0pt}
}
{
	\pgfpagesphysicalpageoptions
	{
		logical pages=4,%
		physical height=\pgfpageoptionheight,%
		physical width=\pgfpageoptionwidth%
	}
	\pgfpageslogicalpageoptions{1}
	{%
		border shrink=\pgfpageoptionborder,%
		resized width=.5\pgfphysicalwidth,%
		resized height=.398\pgfphysicalheight,%
		center=\pgfpoint{.25\pgfphysicalwidth}{.801\pgfphysicalheight}%
	}%
	\pgfpageslogicalpageoptions{3}
	{%
		border shrink=\pgfpageoptionborder,%
		resized width=.5\pgfphysicalwidth,%
		resized height=.398\pgfphysicalheight,%
		center=\pgfpoint{.75\pgfphysicalwidth}{.801\pgfphysicalheight}%
	}%
	\pgfpageslogicalpageoptions{2}
	{%
		border shrink=\pgfpageoptionborder,%
		resized width=.5\pgfphysicalwidth,%
		resized height=.602\pgfphysicalheight,%
		center=\pgfpoint{.25\pgfphysicalwidth}{.301\pgfphysicalheight}%
	}%
	\pgfpageslogicalpageoptions{4}
	{%
		border shrink=\pgfpageoptionborder,%
		resized width=.5\pgfphysicalwidth,%
		resized height=.602\pgfphysicalheight,%
		center=\pgfpoint{.75\pgfphysicalwidth}{.301\pgfphysicalheight}%
	}%
}
